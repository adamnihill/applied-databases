SELECT person.personname, city.name,CASE  WHEN DATEDIFF(hasvisitedcity.dateLeft, hasvisitedcity.dateArrived) < 20 THEN "Short"  WHEN DATEDIFF(hasvisitedcity.dateLeft, hasvisitedcity.dateArrived) BETWEEN 20 AND 99 THEN "Long"  WHEN DATEDIFF(hasvisitedcity.dateLeft, hasvisitedcity.dateArrived) > 99 THEN "Very Long"END as "Stay Length"FROMhasvisitedcityINNER JOIN person ON hasvisitedcity.personID = person.personIDINNER JOIN city ON hasvisitedcity.cityID = city.IDORDER BYperson.personname ASC,city.name ASC;
